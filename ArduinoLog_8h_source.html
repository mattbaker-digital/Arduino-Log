377 template <class T> void printLevel(int level, bool cr, T msg, ...) {
378#ifndef DISABLE_LOGGING
379 if (level > _level) {
380 return;
381 }
382 if (level < LOG_LEVEL_SILENT) {
383 level = LOG_LEVEL_SILENT;
384 }
385
386#ifdef ESP32
387 if (xSemaphoreTake(_semaphore, (TickType_t) 10 )) {
388#endif
389 if (_showColors) {
390 switch (level) {
391 case LOG_LEVEL_FATAL:
392 writeLog(LOG_COLOR_BOLD_RED);
393 break;
394 case LOG_LEVEL_ERROR:
395 writeLog(LOG_COLOR_RED);
396 break;
397 case LOG_LEVEL_WARNING:
398 writeLog(LOG_COLOR_YELLOW);
399 break;
400 default:
401 break;
402 }
403 }
404
405 if (_prefix != NULL && _showLevel) {
406 for (int i = 0; i < _handlerCount; i++) {
407 if (_logOutputs[i]) {
408 _prefix(_logOutputs[i], level);
409 }
410 }
411 }
412
413 if (_showLevel) {
414 static const char levels[] = "FEWITV";
415 writeLog(levels[level - 1]);
416 writeLog(": ");
417 }
418
419 va_list args;
420 va_start(args, msg);
421 print(msg, args);
422
423 if (_suffix != NULL && _showLevel) {
424 for (int i = 0; i < _handlerCount; i++) {
425 if (_logOutputs[i]) {
426 _suffix(_logOutputs[i], level);
427 }
428 }
429 }
430
431 if (cr) {
432 writeLog(CR);
433 }
434
435 if (_showColors) {
436 writeLog(LOG_COLOR_END);
437 }
438
439 va_end(args);
440#ifdef ESP32
441 xSemaphoreGive(_semaphore);
442 }
443#endif
444#endif
445 }

448 int _level;
449 bool _showLevel;
