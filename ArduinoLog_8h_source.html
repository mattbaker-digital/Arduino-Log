376 template <class T> void printLevel(int level, bool cr, T msg, ...) {
377#ifndef DISABLE_LOGGING
378 if (level > _level) {
379 return;
380 }
381 if (level < LOG_LEVEL_SILENT) {
382 level = LOG_LEVEL_SILENT;
383 }
384
385#ifdef ESP32
386 if (xSemaphoreTake(_semaphore, (TickType_t) 10 )) {
387#endif
388 if (_showColors) {
389 switch (level) {
390 case LOG_LEVEL_FATAL:
391 writeLog(LOG_COLOR_BOLD_RED);
392 break;
393 case LOG_LEVEL_ERROR:
394 writeLog(LOG_COLOR_RED);
395 break;
396 case LOG_LEVEL_WARNING:
397 writeLog(LOG_COLOR_YELLOW);
398 break;
399 default:
400 break;
401 }
402 }
403
404 if (_prefix != NULL && _showLevel) {
405 for (int i = 0; i < _handlerCount; i++) {
406 if (_logOutputs[i]) {
407 _prefix(_logOutputs[i], level);
408 }
409 }
410 }
411
412 if (_showLevel) {
413 static const char levels[] = "FEWITV";
414 writeLog(levels[level - 1]);
415 writeLog(": ");
416 }
417
418 va_list args;
419 va_start(args, msg);
420 print(msg, args);
421
422 if (_suffix != NULL && _showLevel) {
423 for (int i = 0; i < _handlerCount; i++) {
424 if (_logOutputs[i]) {
425 _suffix(_logOutputs[i], level);
426 }
427 }
428 }
429
430 if (cr) {
431 writeLog(CR);
432 }
433
434 if (_showColors) {
435 writeLog(LOG_COLOR_END);
436 }
437
438 va_end(args);
439#ifdef ESP32
440 xSemaphoreGive(_semaphore);
441 }
442#endif
443#endif
444 }

448 bool _showLevel;
